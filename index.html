Quick Start

// 1. Add to Package.swift
dependencies: [
    .package(url: "https://github.com/anansadiya94/FeedbackKit", from: "1.0.0")
]

// 2. Basic usage with NoOp provider
import SwiftUI
import FeedbackKit

struct ContentView: View {
    @State private var showFeedback = false

    var body: some View {
        Button("Send Feedback") {
            showFeedback = true
        }
        .sheet(isPresented: $showFeedback) {
            FeedbackView(
                store: Store(initialState: FeedbackFeature.State()) {
                    FeedbackFeature()
                }
            )
        }
    }
}

// 3. With Jira & AI (optional)
import FeedbackKitJira
import FeedbackKitAI

withDependencies {
    $0.feedbackProvider = JiraProvider(configuration: jiraConfig)
    $0.descriptionEnhancer = OpenAIEnhancer(configuration: aiConfig)
}
