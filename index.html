Simple Integration

// 1. Add the dependency
import FeedbackKit

// 2. Present the feedback view
struct MyApp: View {
    @State private var showFeedbackKit = false
    @State private var screenshot: UIImage?

    var body: some View {
        Button("Report Issue") {
            screenshot = ScreenshotHelper.captureWindowImage()
            showFeedbackKit = true
        }
        .sheet(isPresented: $showFeedbackKit) {
            FeedbackKitView(.init(screenshot: screenshot))
        }
        .onReceive(NotificationCenter.default.publisher(
            for: UIApplication.userDidTakeScreenshotNotification
        )) { _ in
            screenshot = ScreenshotHelper.captureWindowImage()
            showFeedbackKit = true
        }
    }
}

// That's it! üéâ
// 1. Add to libs.versions.toml
[versions]
feedbackKit = "1.0.0"

[libraries]
feedback-kit = { group = "com.feedbackkit", name = "sdk", version.ref = "feedbackKit" }

// 2. Add the dependency to build.gradle.kts
dependencies {
    implementation(libs.feedback.kit)
}

// 3. Show feedback screen
class MainActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        setContent {
            var showFeedback by remember { mutableStateOf(false) }

            Button(onClick = { showFeedback = true }) {
                Text("Report Issue")
            }

            if (showFeedback) {
                FeedbackKitView(
                    viewModel = viewModel(),
                    onDismiss = { showFeedback = false }
                )
            }
        }
    }
}
