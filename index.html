Quick Start

// 1. Add to Package.swift
dependencies: [
    .package(url: "https://github.com/anansadiya94/FeedbackKit", from: "1.0.0")
]

// 2. Basic usage with NoOp provider
import SwiftUI
import FeedbackKit

struct ContentView: View {
    @State private var showFeedback = false

    var body: some View {
        Button("Send Feedback") {
            showFeedback = true
        }
        .sheet(isPresented: $showFeedback) {
            FeedbackView(
                store: Store(initialState: FeedbackFeature.State()) {
                    FeedbackFeature()
                }
            )
        }
    }
}

// 3. With Jira & AI (included in single import!)

Jira Integration

// Submit feedback directly to Jira
import SwiftUI
import FeedbackKit  // ‚ú® Includes Core, UI, Jira, and AI!

struct ContentView: View {
    @State private var showFeedback = false

    var body: some View {
        Button("Report an Issue") {
            showFeedback = true
        }
        .sheet(isPresented: $showFeedback) {
            FeedbackView(
                store: Store(initialState: FeedbackFeature.State()) {
                    FeedbackFeature()
                } withDependencies: {
                    $0.feedbackProvider = makeJiraProvider()
                }
            )
        }
    }

    func makeJiraProvider() -> JiraProvider {
        let config = JiraConfiguration(
            baseURL: URL(string: "https://your-company.atlassian.net")!,
            email: ProcessInfo.processInfo.environment["JIRA_EMAIL"]!,
            apiToken: ProcessInfo.processInfo.environment["JIRA_API_TOKEN"]!,
            projectKey: "PROJ",
            issueType: "Bug"
        )
        return JiraProvider(configuration: config)
    }
}

// üéØ No need to import FeedbackKitJira separately!

Perfect for: Production apps that need to track bugs in Jira. Supports custom fields, attachments, and automatic metadata collection.

üí° Tip: Store credentials in environment variables, never hardcode them. Use Xcode schemes or .xcconfig files for different environments.

Full Setup with AI Enhancement

// Jira + AI-powered description enhancement
import SwiftUI
import FeedbackKit  // ‚ú® Everything you need in one import!

struct ContentView: View {
    @State private var showFeedback = false

    var body: some View {
        Button("Send Feedback") {
            showFeedback = true
        }
        .sheet(isPresented: $showFeedback) {
            FeedbackView(
                store: Store(initialState: FeedbackFeature.State()) {
                    FeedbackFeature()
                } withDependencies: {
                    // Configure Jira provider
                    $0.feedbackProvider = JiraProvider(
                        configuration: makeJiraConfig()
                    )

                    // Add AI enhancement (OpenAI or Claude)
                    $0.descriptionEnhancer = OpenAIEnhancer(
                        configuration: .openAI(
                            apiKey: ProcessInfo.processInfo.environment["OPENAI_API_KEY"]!
                        )
                    )

                    // Or use Anthropic Claude instead:
                    // $0.descriptionEnhancer = AnthropicEnhancer(
                    //     configuration: .anthropic(
                    //         apiKey: ProcessInfo.processInfo.environment["ANTHROPIC_API_KEY"]!
                    //     )
                    // )
                }
            )
        }
    }

    func makeJiraConfig() -> JiraConfiguration {
        JiraConfiguration(
            baseURL: URL(string: "https://your-company.atlassian.net")!,
            email: ProcessInfo.processInfo.environment["JIRA_EMAIL"]!,
            apiToken: ProcessInfo.processInfo.environment["JIRA_API_TOKEN"]!,
            projectKey: "PROJ",
            issueType: "Bug"
        )
    }
}

// üöÄ All features available without multiple imports!

Perfect for: Enterprise apps that want AI to improve vague user feedback into clear, actionable bug reports. Transform "app crashes" into detailed descriptions with steps to reproduce.

‚ú® Pro Tip: The AI enhancer runs automatically when users submit feedback. It analyzes the description, adds context from metadata, and improves clarity - all behind the scenes.
